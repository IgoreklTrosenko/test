import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

def index_sum(n_customers):
    number_sum =0
    number=str(n_customers)
    for i in range(len(number)):
        number_sum+=int(number[i])
    return number_sum
        
def build_index(n_customers):
    n =0
    index =[]
    for i in range(n_customers ):
        index.append(n)
        n+=1
        if n == int(n_customers ):
            df=pd.DataFrame(index,columns = ['count'])
    df['count']=df['count'].apply(index_sum)
    count_grooup = df.groupby('count')['count'].count()
    return count_grooup
    
df3 = pd.DataFrame(build_index(761))

plt.figure(figsize=(10,5))
plt.title('Имеет форму нормального распределения')
plt.plot(df3.index,df3.values)
plt.show()
